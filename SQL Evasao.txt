SELECT 
sa.DS_SIT_ALU,
row_number() OVER (ORDER BY CD_ALUNO) AS aluno,
DT_SIT_ALU, CD_ANO_INGRESSO, CD_SEM_INGRESSO,CD_ANO_ENS_MEDIO, CD_SEXO, DS_BAIRRO, DS_CIDADE, DS_ESTADO,
DEFICIENCIA, COR_RACA, num_idade, BOLSAS_COTA_PRIORIDADE, N_COLOCA, N_TOTESC, N_NOTRED
from ACADEMICO.dbo.Alunos al 
inner JOIN ACADEMICO.dbo.SituacoesAlunos sa ON al.CD_SIT_ALU = sa.CD_SIT_ALU 
where al.CD_ANO_INGRESSO = '2022' and al.CD_CURSO like '014';
